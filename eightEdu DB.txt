-- 사용자 테이블
CREATE TABLE USERS(
    USER_CODE NUMBER PRIMARY KEY,
    USER_NAME VARCHAR2(30) NOT NULL,
    USER_ID VARCHAR2(30) NOT NULL,
    USER_PWD VARCHAR2(30) NOT NULL,
    EMAIL VARCHAR2(40),
    PHONE VARCHAR2(30),
    GENDER VARCHAR(2) NOT NULL,
    USER_NO VARCHAR2(30) NOT NULL,
    ADDRESS VARCHAR2(100),
    ENROLL_DATE DATE DEFAULT SYSDATE,
    POINT NUMBER DEFAULT 0,
    GRADE VARCHAR(2) DEFAULT 'C',
    STATUS VARCHAR(2) DEFAULT 'Y',
    ALIASS VARCHAR2(30) NOT NULL
);
-- 말머리 테이블
CREATE TABLE HEAD(
    M_CODE NUMBER PRIMARY KEY,
    M_NAME VARCHAR2(30) NOT NULL
);
-- 게시판 테이블
CREATE TABLE BOARD(
    B_CODE NUMBER PRIMARY KEY,
    USER_CODE NUMBER,
    M_CODE NUMBER,
    CREATE_DATE DATE DEFAULT SYSDATE,
    B_COUNT NUMBER DEFAULT 0,
    B_TITLE VARCHAR2(200) NOT NULL,
    B_CONTENT CLOB NOT NULL,
    STATUS VARCHAR2(2) DEFAULT 'Y',
    B_NAME VARCHAR2(15) NOT NULL,
    CONSTRAINT FK_USER_CODE_BOARD FOREIGN KEY (USER_CODE) REFERENCES USERS(USER_CODE) ON DELETE CASCADE,
    CONSTRAINT FK_M_CODE_BOARD FOREIGN KEY (M_CODE) REFERENCES HEAD (M_CODE) ON DELETE CASCADE
);

-- 강사 테이블
CREATE TABLE TEACHER(
    T_CODE NUMBER PRIMARY KEY,
    T_NAME VARCHAR2(20) NOT NULL,
    T_PHONE VARCHAR2(30),
    T_ADDRESS VARCHAR2(100),
     STATUS VARCHAR2(2) DEFAULT 'Y'

);

-- 강의 리스트 테이블
CREATE TABLE LECTURE(
    L_CODE NUMBER PRIMARY KEY,
    L_TITLE VARCHAR2(500) NOT NULL,
    L_NUMBER NUMBER NOT NULL,
    L_PRICE NUMBER,
    L_TYPE VARCHAR2(100) NOT NULL,
    T_CODE NUMBER NOT NULL,
    L_CONTENT VARCHAR2(4000),
    STATUS VARCHAR2(2) DEFAULT 'Y',
    CONSTRAINT FK_T_CODE_LECLIST FOREIGN KEY(T_CODE) REFERENCES TEACHER (T_CODE) ON DELETE CASCADE
);

-- 게시판 댓글 테이블
CREATE TABLE BOARD_COMMENT(
    C_CODE NUMBER,
    B_CODE NUMBER,
    C_CONTENT VARCHAR2(3000),
    CREATE_DATE DATE DEFAULT SYSDATE,
    STATUS VARCHAR2(2) DEFAULT 'Y',
    USER_CODE NUMBER NOT NULL,
    CONSTRAINT PK_BOARD_COMMENT PRIMARY KEY(C_CODE, B_CODE),
    CONSTRAINT FK_B_CODE_BOARD_COMMENT FOREIGN KEY(B_CODE) REFERENCES BOARD(B_CODE) ON DELETE CASCADE,
    CONSTRAINT FK_USER_CODE_BOARD_COMMENT FOREIGN KEY(USER_CODE) REFERENCES USERS(USER_CODE) ON DELETE CASCADE
);


-- 강좌 테이블
CREATE TABLE CLASS(
    C_CODE NUMBER PRIMARY KEY,
    C_TITLE VARCHAR2(500) NOT NULL,
    C_TIME VARCHAR2(20) NOT NULL,
    C_NEXT NUMBER,
    L_CODE NUMBER,
    T_CODE NUMBER NOT NULL,
    STATUS VARCHAR2(2) DEFAULT 'Y',
    CONSTRAINT FK_T_CODE_CLASS FOREIGN KEY(T_CODE) REFERENCES TEACHER(T_CODE) ON DELETE CASCADE,
    CONSTRAINT FK_L_CODE_CLASS FOREIGN KEY(L_CODE) REFERENCES LECTURE(L_CODE) ON DELETE CASCADE
);

-- 수강 여부 테이블
CREATE TABLE LECYN(
    USER_CODE NUMBER,
    L_CODE NUMBER,
    LECLEVEL VARCHAR2(10) DEFAULT '수강중',
    CONSTRAINT PK_LECYN PRIMARY KEY(USER_CODE, L_CODE),
    CONSTRAINT FK_USER_CODE FOREIGN KEY(USER_CODE) REFERENCES USERS(USER_CODE) ON DELETE CASCADE,
    CONSTRAINT FK_L_CODE FOREIGN KEY(L_CODE) REFERENCES LECTURE (L_CODE) ON DELETE CASCADE
);

-- 게시판 신고 테이블
CREATE TABLE BOARD_REPORT(
    R_CODE NUMBER PRIMARY KEY,
    R_CONTENT VARCHAR2(300) NOT NULL,
    USER_CODE NUMBER NOT NULL,
    B_CODE NUMBER NOT NULL,
    R_NUMBER VARCHAR2(30) NOT NULL,
    CONSTRAINT FK_USER_CODE_BOARD_REPORT FOREIGN KEY(USER_CODE) REFERENCES USERS(USER_CODE) ON DELETE CASCADE,
    CONSTRAINT FK_B_CODE_BOARD_REPORT FOREIGN KEY(B_CODE) REFERENCES BOARD(B_CODE) ON DELETE CASCADE
);
-- 상품 테이블
CREATE TABLE PRODUCT(
    PRO_NUM NUMBER PRIMARY KEY,
    PRO_NAME VARCHAR2(100) NOT NULL,
    PRO_TYPE VARCHAR2(20) NOT NULL,
    REMAIN NUMBER NOT NULL,
    PRICE NUMBER NOT NULL
);
INSERT INTO PRODUCT VALUES(1, 'DD', '노트', 20, 2);
-- 주문내역 테이블(수정)
CREATE TABLE USER_ORDER(
    ORDER_NUM NUMBER PRIMARY KEY,
    USER_CODE NUMBER NOT NULL,
    PRO_NUM NUMBER NOT NULL,
    ORDER_DATE DATE DEFAULT SYSDATE,
    GRADE NUMBER DEFAULT 1,
    CONSTRAINT FK_USER_CODE_USER_ORDER FOREIGN KEY(USER_CODE) REFERENCES USERS(USER_CODE) ON DELETE CASCADE,
    CONSTRAINT FK_PRO_NUM_USER_ORDER FOREIGN KEY(PRO_NUM) REFERENCES PRODUCT(PRO_NUM) ON DELETE CASCADE
);

-- 주문내역(강좌)
CREATE TABLE USER_CLASS(
    ORDER_NUM NUMBER PRIMARY KEY,
    USER_CODE NUMBER NOT NULL,
    L_CODE NUMBER NOT NULL,
    T_CODE NUMBER NOT NULL,
    ORDER_DATE DATE DEFAULT SYSDATE,
    LECLEVEL VARCHAR2(20) DEFAULT '수강중',
    
    CONSTRAINT FK_T_CODE_USER_CLASS FOREIGN KEY(T_CODE) REFERENCES TEACHER(T_CODE) ON DELETE CASCADE,
    CONSTRAINT FK_USER_CODE_USER_CLASS FOREIGN KEY(USER_CODE) REFERENCES USERS(USER_CODE) ON DELETE CASCADE,
    CONSTRAINT FK_L_CODE_USER_CLASS FOREIGN KEY(L_CODE) REFERENCES LECTURE(L_CODE) ON DELETE CASCADE
);



CREATE TABLE NOTICE(
    N_CODE NUMBER PRIMARY KEY,
    USER_CODE NUMBER,
    M_CODE NUMBER,
    CREATE_DATE DATE DEFAULT SYSDATE,
    N_COUNT NUMBER DEFAULT 0,
    N_TITLE VARCHAR2(200) NOT NULL,
    N_CONTENT CLOB NOT NULL,
    STATUS VARCHAR2(2) DEFAULT 'Y',
    N_NAME VARCHAR2(20) NOT NULL,
    CONSTRAINT FK_USER_CODE_NOTICE FOREIGN KEY (USER_CODE) REFERENCES USERS(USER_CODE) ON DELETE CASCADE,
    CONSTRAINT FK_M_CODE_NOTICE FOREIGN KEY (M_CODE) REFERENCES HEAD (M_CODE) ON DELETE CASCADE
); 

-- 동영상
CREATE TABLE VIDEO(
    V_CODE NUMBER PRIMARY KEY,
    V_LINK VARCHAR2(300) NOT NULL,
    C_CODE NUMBER NOT NULL,
     STATUS VARCHAR2(2) DEFAULT 'Y',
    CONSTRAINT FK_C_CODE_VIDEO FOREIGN KEY (C_CODE) REFERENCES CLASS (C_CODE) ON DELETE CASCADE
);


-- 공지사항 댓글
CREATE TABLE NOTICE_COMMENT(
    NC_CODE NUMBER,
    N_CODE NUMBER,
    NC_CONTENT VARCHAR2(3000),
    CREATE_DATE DATE DEFAULT SYSDATE,
    STATUS CHAR(2) DEFAULT 'Y',
    USER_CODE NUMBER NOT NULL,
    CONSTRAINT PK_NOTICE_COMMENT PRIMARY KEY(NC_CODE, N_CODE),
    CONSTRAINT FK_N_CODE FOREIGN KEY(N_CODE) REFERENCES NOTICE(N_CODE) ON DELETE CASCADE,
    CONSTRAINT FK_USER_CODE_NOTICE_COMMENT FOREIGN KEY(USER_CODE) REFERENCES USERS(USER_CODE) ON DELETE CASCADE
);

  CREATE OR REPLACE FORCE VIEW "EIGHT"."BDETAIL" ("B_CODE", "USER_CODE", "M_CODE", "CREATE_DATE", "B_COUNT", "B_TITLE", "B_CONTENT", "STATUS", "B_NAME", "M_NAME", "ALIASS") AS 
  SELECT B_CODE,B.USER_CODE,B.M_CODE,CREATE_DATE,B_COUNT,B_TITLE,B_CONTENT,B.STATUS,B_NAME,M_NAME,ALIASS
FROM BOARD B
JOIN USERS U ON(B.USER_CODE = U.USER_CODE)
JOIN HEAD H ON(B.M_CODE = H.M_CODE)
WHERE B.STATUS = 'Y';

  CREATE OR REPLACE FORCE VIEW "EIGHT"."BLIST" ("B_CODE", "USER_CODE", "M_CODE", "CREATE_DATE", "B_COUNT", "B_TITLE", "B_CONTENT", "STATUS", "B_NAME", "M_NAME", "ALIASS") AS 
  SELECT B_CODE, USER_CODE, M_CODE, CREATE_DATE, B_COUNT, B_TITLE, B_CONTENT, B.STATUS, B_NAME, M_NAME, ALIASS 
FROM BOARD B
JOIN HEAD USING(M_CODE) 
JOIN USERS USING(USER_CODE)
WHERE B.STATUS='Y' 
ORDER BY B_CODE DESC;

  CREATE OR REPLACE FORCE VIEW "EIGHT"."BULIST" ("RNUM", "USER_CODE", "USER_NO", "USER_ID", "USER_NAME", "EMAIL", "PHONE", "ADDRESS", "ENROLL_DATE") AS 
  SELECT ROWNUM RNUM, A."USER_CODE",A."USER_NO",A."USER_ID",A."USER_NAME",A."EMAIL",A."PHONE",A."ADDRESS",A."ENROLL_DATE"
FROM (SELECT USER_CODE, USER_NO, USER_ID, USER_NAME, EMAIL, PHONE, ADDRESS, ENROLL_DATE
FROM USERS U
WHERE STATUS='N'
ORDER BY USER_CODE DESC)A;

  CREATE OR REPLACE FORCE VIEW "EIGHT"."CLIST" ("RNUM", "C_CODE", "C_TITLE", "C_TIME", "C_NEXT", "L_CODE", "T_CODE", "STATUS", "T_NAME", "L_TITLE") AS 
  SELECT ROWNUM RNUM, A."C_CODE",A."C_TITLE",A."C_TIME",A."C_NEXT",A."L_CODE",A."T_CODE",A."STATUS",A."T_NAME",A."L_TITLE"
FROM(SELECT C_CODE, C_TITLE, C_TIME, C_NEXT, L_CODE, A.T_CODE, A.STATUS, T_NAME, L_TITLE
FROM CLASS A
JOIN LECTURE B USING(L_CODE)
JOIN TEACHER C ON A.T_CODE = C.T_CODE
WHERE A.STATUS='Y'
ORDER BY C_CODE DESC)A;

  CREATE OR REPLACE FORCE VIEW "EIGHT"."CLIST_1" ("C_CODE", "C_TITLE", "C_TIME", "C_NEXT", "L_CODE", "T_CODE", "STATUS", "T_NAME", "L_TITLE") AS

  CREATE OR REPLACE FORCE VIEW "EIGHT"."MYBUYLIST" ("PRO_NAME", "PRO_TYPE", "PRICE", "ORDER_DATE", "USER_CODE") AS
